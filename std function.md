
```cpp
// Определение структуры S
struct S {
    int c; // Поле структуры, которое будет хранить значение целого типа

    // Метод структуры, который умножает аргумент на значение поля c
    int mult(int arg) {
        return arg * c; // Возвращает произведение аргумента и значения поля c
    }
};

// Определение std::function, которая хранит указатель на метод mult структуры S
std::function<int(S&, int)> func = &S::mult;

// Создание экземпляра структуры S с инициализацией поля c значением 10
S s{10};

// Вызов метода mult через std::function, передавая объект s и аргумент 3
func(s, 3); // Здесь func вызывает метод mult для объекта s, умножая 3 на 10, результат будет 30

// Получение указателя на метод mult структуры S
auto ptr = &S::mult;

// Вызов метода mult через указатель на метод, используя объект s и аргумент 3
(s.*ptr)(3); // Здесь s.*ptr вызывает метод mult для объекта s, умножая 3 на 10, результат будет 30
```
